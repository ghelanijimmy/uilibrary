## Single Paragraph component

export const Template = ({
  size,
  children,
  color,
  asSpan,
  displayInline,
  makeInlineBlock,
  makeBlock,
}) => (
  <Paragraph
    size={size}
    color={color}
    asSpan={asSpan}
    displayInline={displayInline}
    makeInlineBlock={makeInlineBlock}
    makeBlock={makeBlock}
  >
    {children}
  </Paragraph>
);

export const DefaultParagraph = Template.bind({});
DefaultParagraph.args = {
  size: MD,
  color: "",
  asSpan: false,
  displayInline: false,
  makeInlineBlock: false,
  makeBlock: false,
};
